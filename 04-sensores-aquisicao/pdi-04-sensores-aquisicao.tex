    \begin{slide}[toc=]{Modelo de formação de imagem}
       \begin{equation*}
          f(x,y) = i(x,y)r(x,y) 
       \end{equation*}
      \begin{align*}
         0<&f(x,y)<\infty\\
         0<&i(x,y)<\infty\\
         0<&r(x,y)<1
      \end{align*}
      \begin{itemize}
       \item $f(x,y)$: intensidade
       \item $i(x,y)$: iluminação
       \item $r(x,y)$: refletância
      \end{itemize}
    \end{slide}
